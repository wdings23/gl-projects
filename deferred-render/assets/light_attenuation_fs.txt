uniform sampler2D posTex;

varying vec4 vColor;
varying vec2 vTexCoord;
varying vec4 vLightPos;
varying vec4 vPosition;

void main( void )
{
	float fDP = dot( vPosition.xy, vPosition.xy );
	float fPct = sqrt( fDP * fDP );

	gl_FragColor.xyz = vColor.xyz;
	gl_FragColor.w = 0.25f; // ( 1.0 - fPct ) * 0.5;
}
